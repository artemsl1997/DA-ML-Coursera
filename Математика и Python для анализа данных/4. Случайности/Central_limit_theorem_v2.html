def norm_dist_plot(selection):
    avg = np.array([])
    number_of_selections = 1000
    for i in range(number_of_selections):
        l = our_laplace.rvs(size=selection)
        avg = np.append(avg, l.mean())
        
    #    
    std = (var_l / selection)**0.5
    mean = mean_l
    
    print(" . : {};\n . : {}.".format(mean, round(std, 3)))
    
    plt.figure()
    plt.hist(avg, bins=20, density=True, color='b', label='. .')
    plt.plot(np.sort(avg), norm.pdf(np.sort(avg), mean, std), 'r', lw=4, label=' . .')
    plt.grid()
    plt.title('    1000    n = {}'.format(selection))
    plt.legend(loc='best', frameon=True)
    plt.xlabel('x')
    plt.ylabel('f(x)')
    plt.show()
