def norm_dist_plot(selection):
    avg = np.array([])
    number_of_selections = 1000
    for i in range(number_of_selections):
        l = our_laplace.rvs(size=selection)
        avg = np.append(avg, l.mean())
        
    # расчет параметров нормального распределения
    std = (var_l / selection)**0.5
    mean = mean_l
    
    print("Матожидание норм. распределения: {};\nСКО норм. распределения: {}.".format(mean, round(std, 3)))
    
    plt.figure()
    plt.hist(avg, bins=20, density=True, color='b', label='средневыб. знач.')
    plt.plot(np.sort(avg), norm.pdf(np.sort(avg), mean, std), 'r', lw=4, label='плотность норм. распр.')
    plt.grid()
    plt.title('Гистограмма выборочных средних из 1000 выборок с объемом n = {}'.format(selection))
    plt.legend(loc='best', frameon=True)
    plt.xlabel('x')
    plt.ylabel('f(x)')
    plt.show()

Выводы¶

  1. На основе созданных выборок разного объема (от 5 до 500) была проверена работа центральной предельной теоремы;
  2. При объеме выборки > 10 СКО нормального распределения меньше 0.5
